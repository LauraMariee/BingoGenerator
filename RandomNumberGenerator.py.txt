# generate random integer values from random import seed
from random import randint

calledNumbers = []


def main():
    # generate a integers
    numCheck()
        

def numCheck():
    value = randint(1, 50)
    if value in calledNumbers:
        print("Number has been called!")
        numCheck()
    else:
        calledNumbers.append(value)
        print(value)

if __name__ == "__main__":
    main()
